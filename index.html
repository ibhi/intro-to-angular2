Template Syntax 

						
@Component({
  selector: 'name-change'
})
@View({
  templateUrl: '/name-change.html'
})
class NameChange {
  name: string;
  constructor() {
    this.name = '';
  }
  changeName(newName) {
    this.name = newName;
  }
}
						
					
						
<div> My name is {{ name }} </div>
<div>
  <input #newname type="text">
  <button (click)="changeName(newname.value)"
          [disabled]="newname.value == 'Ibhi'"> Change Name
  </button>
</div>
